      <button onclick="startScanner()">Start Camera</button>

      <button class="secondary" onclick="stopScanner()">Stop Camera</button>


      <input id="manualUrl" placeholder="Paste a QR URL manually if needed" />

      <button class="secondary" onclick="openManualUrl()">Open QR URL</button>

  <script>

    let scanner = null;

    let running = false;

    let lastScan = '';

    let lastScanAt = 0;


    function setPanel(title, msg, isError = false) {

      const panel = document.getElementById('panel');

      panel.className = 'panel' + (isError ? ' error' : '');

      document.getElementById('title').textContent = title || 'Ready';

      document.getElementById('msg').textContent = msg || '';

    }


    async function startScanner() {

      if (running) return;


      try {

        scanner = new Html5Qrcode('reader');

        await scanner.start(

          { facingMode: 'environment' },

          { fps: 10, qrbox: { width: 240, height: 240 } },

          async (decodedText) => {

            const now = Date.now();

            if (decodedText === lastScan && now - lastScanAt < 2500) return;

            lastScan = decodedText;

            lastScanAt = now;


            if (!decodedText.startsWith('http://') && !decodedText.startsWith('https://')) {

              setPanel('Invalid QR', 'This QR does not contain a URL.', true);

              return;

            }


            setPanel('Opening Check-in', 'Sending to Apps Script...');

            window.location.href = decodedText;

          }

        );


        running = true;

        setPanel('Scanner Live', 'Point camera at badge QR.');

      } catch (err) {

        setPanel('Camera Error', err.message || 'Camera permission failed.', true);

      }

    }


    async function stopScanner() {

      try {

        if (scanner && running) {

          await scanner.stop();

          await scanner.clear();

        }

      } catch (e) {}

      scanner = null;

      running = false;

      setPanel('Stopped', 'Camera stopped.');

    }


    function openManualUrl() {

      const url = document.getElementById('manualUrl').value.trim();

      if (!url) {

        setPanel('Missing URL', 'Paste a QR URL first.', true);

        return;

      }

      window.location.href = url;

    }
