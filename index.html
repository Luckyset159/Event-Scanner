  <script>

    let scanner = null;

    let running = false;

    let lastScan = '';

    let lastScanAt = 0;


    function setPanel(type, title, msg, attendee) {

      const panel = document.getElementById('panel');

      panel.className = 'panel';

      if (type === 'success') panel.classList.add('success');

      if (type === 'duplicate') panel.classList.add('duplicate');

      if (type === 'error' || type === 'invalid') panel.classList.add('error');


      document.getElementById('title').textContent = title || 'Ready';

      document.getElementById('msg').textContent = msg || '';


      const name = document.getElementById('name');

      const post = document.getElementById('post');


      if (attendee && attendee.name) {

        name.style.display = '';

        post.style.display = '';

        name.textContent = attendee.name;

        post.textContent = attendee.post || '';

      } else {

        name.style.display = 'none';

        post.style.display = 'none';

        name.textContent = '';

        post.textContent = '';

      }

    }


    function extractToken(scannedText) {

      try {

        if (scannedText.startsWith('http://') || scannedText.startsWith('https://')) {

          const url = new URL(scannedText);

          return url.searchParams.get('token') || '';

        }

      } catch (e) {}

      return scannedText.trim();

    }


    async function sendCheckIn(token) {

      const endpoint = document.getElementById('endpoint').value.trim();

      const checker = document.getElementById('checker').value.trim() || 'Gate Security';

      const key = document.getElementById('eventKey').value.trim();


      if (!endpoint) {

        setPanel('error', 'Missing Endpoint', 'Paste your Apps Script Web App URL first.');

        return;

      }


      setPanel('', 'Checking In...', 'Please wait...');


      try {

        const res = await fetch(endpoint, {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({

            action: 'apiCheckin',

            token,

            checker,

            key

          })

        });


        const data = await res.json();

        setPanel(data.status, data.title, data.message, data.attendee);

      } catch (err) {

        setPanel('error', 'Network Error', err.message || 'Check connection and endpoint URL.');

      }

    }


    async function startScanner() {

      if (running) return;


      try {

        scanner = new Html5Qrcode('reader');

        await scanner.start(

          { facingMode: 'environment' },

          { fps: 10, qrbox: { width: 240, height: 240 } },

          async (decodedText) => {

            const now = Date.now();

            if (decodedText === lastScan && now - lastScanAt < 2500) return;

            lastScan = decodedText;

            lastScanAt = now;


            const token = extractToken(decodedText);

            if (!token) {

              setPanel('error', 'Invalid QR', 'Could not read token from QR.');

              return;

            }


            await sendCheckIn(token);

          }

        );

        running = true;

        setPanel('', 'Scanner Live', 'Point camera at badge QR.');

      } catch (err) {

        setPanel('error', 'Camera Error', err.message || 'Camera permission failed.');

      }

    }


    async function stopScanner() {

      try {

        if (scanner && running) {

          await scanner.stop();

          await scanner.clear();

        }

      } catch (e) {}

      scanner = null;

      running = false;

      setPanel('', 'Stopped', 'Camera stopped.');

    }


    async function manualCheckIn() {

      const token = document.getElementById('manualToken').value.trim();

      if (!token) {

        setPanel('error', 'Missing Token', 'Enter a token first.');

        return;

      }

      await sendCheckIn(token);

      document.getElementById('manualToken').value = '';

    }
